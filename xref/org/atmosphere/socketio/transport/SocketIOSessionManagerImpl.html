63      private static String generateRandomString(int length) {
64  
65          byte[] bytes = new byte[16];
66  
67          // Render the result as a String of hexadecimal digits
68          StringBuilder buffer = new StringBuilder(length);
69  
70          int resultLenBytes = 0;
71  
72          while (resultLenBytes < length) {
73              random.nextBytes(bytes);
74              for (int j = 0;
75                   j < bytes.length && resultLenBytes < length;
76                   j++) {
77                  byte b1 = (byte) ((bytes[j] & 0xf0) >> 4);
78                  byte b2 = (byte) (bytes[j] & 0x0f);
79                  if (b1 < 10)
80                      buffer.append((char) ('0' + b1));
81                  else
82                      buffer.append((char) ('A' + (b1 - 10)));
83                  if (b2 < 10)
84                      buffer.append((char) ('0' + b2));
85                  else
86                      buffer.append((char) ('A' + (b2 - 10)));
87                  resultLenBytes++;
88              }
89          }
90  
91          return buffer.toString();
92  
93  
94      }

290         @Override
291         public void onClose(String data) {
292             if (state == ConnectionState.CLOSING) {
293                 if (closeId != null && closeId.equals(data)) {
294                     state = ConnectionState.CLOSED;
295                     onDisconnect(DisconnectReason.CLOSED);
296                     handler.abort();
297                 } else {
298                     try {
299                         handler.sendMessage(data);
300                     } catch (SocketIOException e) {
301                         logger.error("handler.sendMessage failed: ", e);
302                         handler.abort();
303                     }
304                 }
305             } else {
306                 clearTimeoutTimer();
307                 clearHeartbeatTimer();
308                 state = ConnectionState.CLOSING;
309                 try {
310                     handler.sendMessage(data);
311                 } catch (SocketIOException e) {
312                     logger.error("handler.sendMessage failed: ", e);
313                     handler.abort();
314                 }
315             }
316         }

318         @Override
319         public void onConnect(AtmosphereResourceImpl resource, SocketIOSessionOutbound handler) {
320             if (handler == null) {
321                 state = ConnectionState.CLOSED;
322                 atmosphereHandler = null;
323                 socketIOSessions.remove(sessionId);
324             } else if (this.handler == null) {
325                 this.handler = handler;
326 
327                 try {
328                     state = ConnectionState.CONNECTED;
329 
330                     if (atmosphereHandler == null) {
331                         // Something went wrong
332                         logger.debug("Invalid state");
333                         return;
334                     }
335 
336                     // for the Broadcaster
337                     resource.getRequest().setAttribute(SocketIOAtmosphereHandler.SOCKETIO_SESSION_ID, sessionId);
338                     resource.getRequest().setAttribute(SocketIOAtmosphereHandler.SOCKETIO_SESSION_OUTBOUND, handler);
339 
340                     startHeartbeatTimer();
341                     synchronized (atmosphereHandler) {
342                         if (SocketIOAtmosphereHandler.class.isAssignableFrom(atmosphereHandler.getClass())) {
343                             SocketIOAtmosphereHandler.class.cast(atmosphereHandler).onConnect(resource, handler);
344                         } else {
345                             resource.disableSuspend(true);
346                             atmosphereHandler.onRequest(resource);
347                         }
348                     }
349                 } catch (Throwable e) {
350                     logger.error("Session[" + sessionId + "]: Exception thrown by SocketIOInbound.onConnect()", e);
351                     state = ConnectionState.CLOSED;
352                     handler.abort();
353                 }
354             } else {
355                 handler.abort();
356             }
357         }

359         @Override
360         public void onMessage(AtmosphereResourceImpl resource, SocketIOSessionOutbound outbound, String message) {
361             startHeartbeatTimer();
362 
363             if (atmosphereHandler != null && message != null) {
364                 try {
365                     synchronized (atmosphereHandler) {
366                         if (SocketIOAtmosphereHandler.class.isAssignableFrom(atmosphereHandler.getClass())) {
367                             SocketIOAtmosphereHandler.class.cast(atmosphereHandler).onMessage(resource, outbound, message);
368                         } else {
369                             SocketIOProtocol p = mapper.readValue(message, SocketIOProtocol.class);
370 
371                             for (String msg : p.getArgs()) {
372                                 AtmosphereRequest r = resource.getRequest();
373                                 r.setAttribute(SocketIOProtocol.class.getName(), p);
374                                 r.body(msg).method("POST");
375                                 resource.disableSuspend(true);
376                                 atmosphereHandler.onRequest(resource);
377                             }
378                         }
379                     }
380                 } catch (Throwable e) {
381                     logger.error("Session[" + sessionId + "]: Exception thrown by SocketIOInbound.onMessage()", e);
382                 }
383             }
384         }

386         @Override
387         public void onDisconnect(DisconnectReason reason) {
388             logger.trace("Session[" + sessionId + "]: onDisconnect: " + reason);
389             clearTimeoutTimer();
390             clearHeartbeatTimer();
391             if (atmosphereHandler != null) {
392                 state = ConnectionState.CLOSED;
393                 try {
394                     synchronized (atmosphereHandler) {
395                         if (SocketIOAtmosphereHandler.class.isAssignableFrom(atmosphereHandler.getClass())) {
396                             SocketIOAtmosphereHandler.class.cast(atmosphereHandler).onDisconnect(resource, handler, reason);
397                         } else {
398                             atmosphereHandler.onStateChange(new AtmosphereResourceEventImpl(resource, true, false));
399                         }
400                     }
401                 } catch (Throwable e) {
402                     logger.error("Session[" + sessionId + "]: Exception thrown by SocketIOInbound.onDisconnect()", e);
403                 }
404                 atmosphereHandler = null;
405             }
406         }

408         @Override
409         public void onShutdown() {
410             logger.trace("Session[" + sessionId + "]: onShutdown");
411             if (atmosphereHandler != null) {
412                 if (state == ConnectionState.CLOSING) {
413                     if (closeId != null) {
414                         onDisconnect(DisconnectReason.CLOSE_FAILED);
415                     } else {
416                         onDisconnect(DisconnectReason.CLOSED_REMOTELY);
417                     }
418                 } else {
419                     onDisconnect(DisconnectReason.ERROR);
420                 }
421             }
422             socketIOSessions.remove(sessionId);
423         }

437     /**
438      * Connection state based on Jetty for the connection's state
439      *
440      * @author Sebastien Dionne  : sebastien.dionne@gmail.com
441      */
442     public enum ConnectionState {
443         UNKNOWN(-1),
444         CONNECTING(0),
445         CONNECTED(1),
446         CLOSING(2),
447         CLOSED(3);
448 
449         private int value;
450 
451         private ConnectionState(int v) {
452             this.value = v;
453         }
454 
455         public int value() {
456             return value;
457         }
458 
459         public static ConnectionState fromInt(int val) {
460             switch (val) {
461                 case 0:
462                     return CONNECTING;
463                 case 1:
464                     return CONNECTED;
465                 case 2:
466                     return CLOSING;
467                 case 3:
468                     return CLOSED;
469                 default:
470                     return UNKNOWN;
471             }
472         }
473     }
