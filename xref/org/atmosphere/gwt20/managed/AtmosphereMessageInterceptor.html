16  package org.atmosphere.gwt20.managed;
17  
18  import org.atmosphere.cpr.Action;
19  import org.atmosphere.cpr.AtmosphereConfig;
20  import org.atmosphere.cpr.AtmosphereInterceptorAdapter;
21  import org.atmosphere.cpr.AtmosphereResource;
22  import org.atmosphere.gwt20.client.AtmosphereMessage;
23  import org.atmosphere.gwt20.shared.Constants;
24  
25  /**
26   * De-serialize GWT object into a String so we can use {@link org.atmosphere.cpr.Broadcaster} and non GWT application as client.
27   * All AtmosphereInterceptor will work out of the box without requiring any change.
28   * <p/>
29   * This interceptor is normally used with the {@link org.atmosphere.config.service.ManagedService} annotation
30   * and must always be used with {@link org.atmosphere.gwt20.server.GwtRpcInterceptor}.
31   *
32   * @author Jeanfrancois Arcand
33   */
34  public class AtmosphereMessageInterceptor extends AtmosphereInterceptorAdapter {
35  
36      private final AtmosphereMessageFilterEncoder encoder = new AtmosphereMessageFilterEncoder();
37  
38      @Override
39      public void configure(AtmosphereConfig config) {
40          config.framework().broadcasterFilters(encoder);
41      }
42  
43      @Override
44      public Action inspect(AtmosphereResource r) {
45          Object msg = r.getRequest().getAttribute(Constants.MESSAGE_OBJECT);
46          if (msg != null && AtmosphereMessage.class.isAssignableFrom(msg.getClass())) {
47              AtmosphereMessage<?> m = AtmosphereMessage.class.cast(msg);
48              encoder.classToEncode((Class<? extends AtmosphereMessage<?>>) m.getClass());
49              if (m.type().equals(AtmosphereMessage.TYPE.STRING)) {
50                  r.getRequest().body(m.asString());
51              } else {
52                  r.getRequest().body(m.asByte());
53              }
54          }
55          return Action.CONTINUE;
56      }
57  }
