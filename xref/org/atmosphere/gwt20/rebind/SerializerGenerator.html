54  import java.io.Serializable;
55  import java.util.ArrayList;
56  import java.util.Collections;
57  import java.util.List;

59  import org.atmosphere.gwt20.client.GwtRpcSerialTypes;

61  public class SerializerGenerator extends IncrementalGenerator {
62  
63      @Override
64      public long getVersionId() {
65          return 1L;
66      }
67  
68      @Override
69      public RebindResult generateIncrementally(TreeLogger logger, GeneratorContext context, String typeName) throws UnableToCompleteException {
70  
71          TypeOracle typeOracle = context.getTypeOracle();
72  
73          // Create the CometSerializer impl
74          String packageName = "comet";
75          String className = typeName.replace('.', '_') + "Impl";
76          PrintWriter printWriter = context.tryCreate(logger, packageName, className);
77  
78          if (printWriter != null) {
79  
80              try {
81                  JClassType type = typeOracle.getType(typeName);
82                  GwtRpcSerialTypes annotation = type.getAnnotation(GwtRpcSerialTypes.class);
83                  if (annotation == null) {
84                      logger.log(TreeLogger.ERROR, "No SerialTypes annotation on CometSerializer type: " + typeName);
85                      throw new UnableToCompleteException();
86                  }
87  
88                  SerializableTypeOracleBuilder typesSentToBrowserBuilder = new SerializableTypeOracleBuilder(
89                          logger, context.getPropertyOracle(), context);
90                  SerializableTypeOracleBuilder typesSentFromBrowserBuilder = new SerializableTypeOracleBuilder(
91                          logger, context.getPropertyOracle(), context);
92  
93                  List<Class<?>> serializableTypes = new ArrayList();
94                  Collections.addAll(serializableTypes, annotation.value());
95                  for (Class<?> serializable : serializableTypes) {
96                      int rank = 0;
97                      if (serializable.isArray()) {
98                          while (serializable.isArray()) {
99                              serializable = (Class<?>) serializable.getComponentType();
100                             rank++;
101                         }
102                     }
103 
104                     JType resolvedType = typeOracle.getType(serializable.getCanonicalName());
105                     while (rank > 0) {
106                         resolvedType = typeOracle.getArrayType(resolvedType);
107                         rank--;
108                     }
109 
110                     typesSentToBrowserBuilder.addRootType(logger, resolvedType);
111                     typesSentFromBrowserBuilder.addRootType(logger, resolvedType);
112                 }
113 
114                 // Create a resource file to receive all of the serialization information
115                 // computed by STOB and mark it as private so it does not end up in the
116                 // output.
117                 OutputStream pathInfo = context.tryCreateResource(logger, typeName + ".rpc.log");
118                 PrintWriter writer = new PrintWriter(new OutputStreamWriter(pathInfo));
119                 writer.write("====================================\n");
120                 writer.write("Types potentially sent from server:\n");
121                 writer.write("====================================\n\n");
122                 writer.flush();
123 
124                 typesSentToBrowserBuilder.setLogOutputWriter(writer);
125                 SerializableTypeOracle typesSentToBrowser = typesSentToBrowserBuilder.build(logger);
126 
127                 writer.write("===================================\n");
128                 writer.write("Types potentially sent from browser:\n");
129                 writer.write("===================================\n\n");
130                 writer.flush();
131                 typesSentFromBrowserBuilder.setLogOutputWriter(writer);
132                 SerializableTypeOracle typesSentFromBrowser = typesSentFromBrowserBuilder.build(logger);
133 
134                 writer.close();
135 
136                 if (pathInfo != null) {
137                     context.commitResource(logger, pathInfo).setPrivate(true);
138                 }
139 
140                 // Create the serializer
141                 final String modifiedTypeName = typeName.replace('.', '_');
142                 TypeSerializerCreator tsc = new TypeSerializerCreator(logger, typesSentFromBrowser, typesSentToBrowser, context, "comet." + modifiedTypeName, modifiedTypeName);
143                 String realize = tsc.realize(logger);
144 
145                 // Create the CometSerializer impl
146                 ClassSourceFileComposerFactory composerFactory = new ClassSourceFileComposerFactory(packageName, className);
147 
148                 composerFactory.addImport(Serializer.class.getName());
149                 composerFactory.addImport(SerializationException.class.getName());
150                 composerFactory.addImport(Serializable.class.getName());
151 
152                 composerFactory.setSuperclass(typeName);
153                 SourceWriter sourceWriter = composerFactory.createSourceWriter(context, printWriter);
154                 sourceWriter.print("private Serializer SERIALIZER = new " + realize + "();");
155                 sourceWriter.print("protected Serializer getRPCSerializer() {return SERIALIZER;}");
156                 sourceWriter.commit(logger);
157 
158             } catch (NotFoundException e) {
159                 logger.log(TreeLogger.ERROR, "", e);
160                 throw new UnableToCompleteException();
161             }
162         }
163 
164         return new RebindResult(RebindMode.USE_ALL_NEW_WITH_NO_CACHING, packageName + '.' + className);
165     }
